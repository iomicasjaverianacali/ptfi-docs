    def read_ms_from_excel(self, dir_path="./data", filename="Pool moleculas de prueba.xlsx", index_excel_sheet=0, index_ms2_col=-1, index_smi=0):
        """
        Read an Excel file with (potentially) multiple SMILES and Mass spectrums.

        The excel file has ONLY ONE sheet with the following format:

        SMILES,     Precursor m/z,      Name,   MSMS spectrum
        struct1,        1221,    namemolecule1, 12.09 91;13 145;....
        struct2,          12,    namemolecule2, 11.07 76;17.1 232;....

        Args:
            dir_path (str, optional): path to directory containing the excel file. Defaults to "./data".
            filename (str, optional): name of the excel file. Defaults to "Pool moleculas de prueba.xlsx".

        Returns:
            MZ (list of lists): Has the m/z values. Its length corresponds to the number of molecules in the file. The length of each entry of the list is the number of samples in the Mass spectrum.
            ITZ (list of lists): Has the Intentsity values. Its length corresponds to the number of molecules in the file. The length of each entry of the list is the number of samples in the Mass spectrum.
        """

        # Get the Excel sheet
        file_path = os.path.join(dir_path,filename)
        workbook = load_workbook(filename=file_path)
        sheet = workbook[workbook.sheetnames[index_excel_sheet]]

        # Get the columns
        columns = []
        for col in sheet.iter_cols(values_only=True):
            column_data = list(col)
            columns.append(column_data)

        # Get the Mass spectrum
        MZ = []
        ITZ = []

        for ms2 in columns[index_ms2_col][1:]:
            aux = ms2.split(';')
            mz = []
            itz = []
            for entry in aux:
                aux2 = entry.split(' ')
                mz.append(float(aux2[0]))
                itz.append(float(aux2[1]))
            MZ.append(mz)
            ITZ.append(itz)

        smiles = columns[index_smi][1:]

        return MZ, ITZ, smiles

    def make_clean_smiles(self, mz, smiles, prid_num=100):
        judge = 0
        SS = []
        jihe = []
        enc_input2 = []
        dec_input2 = []
        real_smiles = []
        for i in range(1):
            mz = [float(x) for x in mz]
            mz = [int(100 * x) for x in mz]
            if max(mz) > 50000:
                continue
            if len(smiles) < 100:
                for j in range(len(smiles)):
                    if judge == 1:
                        judge = 0
                    else:
                        if smiles[j] == 'C' and j < len(smiles) - 1:
                            if smiles[j + 1] == 'l':
                                SS.append('Cl')
                                judge = 1
                                continue
                        if smiles[j] == 'B' and j < len(smiles) - 1:
                            if smiles[j + 1] == 'r':
                                SS.append('Br')
                                judge = 1
                                continue
                        if smiles[j] == '\xa0': # detects a space at the end of the line in the string
                            continue
                        SS.append(smiles[j])
                SS = ['<SOS>'] + SS + ['<EOS>']
                for j in range(0, 100):
                    mz.append(0)
                    SS.append('<PAD>')
                enc_input = [[self.vocab_mz[n] for n in mz]]
                dec_input = [[self.vocab_smi[n] for n in SS]]
                enc_input2.append(enc_input[0][0:100])
                dec_input2.append(dec_input[0][0:101])
                real_smiles.append(smiles)
                SS = []

        return enc_input2, dec_input2, real_smiles

    def load_data_in_three_energy_levels_from_databases(self, filepath="/home/julian/Documents/Librerias publicas/", he=36, me=16, save_info=False):
        """
        This method loads all the compounds available in the files allocated under `filepath`, and only returns the compounds for which there is a mass spectra in each of the three levels of collision energy. 

        These files correspond to database `.msp` files containing information of the compound such as its name, precursormz, mass spectra, smiles, inkikey, among others. 

        There are repeated compounds. The criterion to filter repeated compounds (choose one of them) is to pick the one having the more data in the mass spectra.

        Args:
            filepath (str, optional): path to the directory under which are allocated all the files containing compounds information. Defaults to "/home/julian/Documents/Librerias publicas/".
            he (int, optional): cutoff value above which are the high level collision energy compounds. Defaults to 36.
            me (int, optional): cutoff value above which are the medium level collision energy compounds. Defaults to 16.
            save_info (bool, optional): if true saves some useful information about all the scraped database files. Defaults to False.

        Returns:
            le_compounds_in_three_energy (list of dicts): each entry of the list is a dictionary having information of the compound such as its name, smiles, inkikey and mass spectra for a low-level collision energy.
            me_compounds_in_three_energy (list of dicts): each entry of the list is a dictionary having information of the compound such as its name, smiles, inkikey and mass spectra for a medium-level collision energy.
            he_compounds_in_three_energy (list of dicts): each entry of the list is a dictionary having information of the compound such as its name, smiles, inkikey and mass spectra for a high-level collision energy. 
        """

        he_compounds = []
        me_compounds = []
        le_compounds = []

        total_compounds_count = 0
        total_compounds_have_energy_info = 0

        for file_i in os.listdir(filepath):
            aux = file_i.split(".")
            if aux[1] == 'npy':
                this_database = np.load(os.path.join(filepath, file_i), allow_pickle=True)
                for entry in this_database:
                    for key, val in entry.items():
                        if key == "COLLISIONENERGY" and val != '' and val != "--":
                            aux2 = val.split("HCD")
                            if len(aux2) == 1:
                                val = val.split(" ")
                                val = float(val[0])
                            else:
                                val = float(aux2[0])
                            if val > 0 and val < me:
                                le_compounds.append(entry)
                            if val >= me and val < he:
                                me_compounds.append(entry)
                            if val >= he:
                                he_compounds.append(entry)

                            total_compounds_have_energy_info = total_compounds_have_energy_info + 1

                    total_compounds_count = total_compounds_count + 1

        names_le = {d["NAME"] for d in le_compounds}
        names_me = {d["NAME"] for d in me_compounds}
        names_he = {d["NAME"] for d in he_compounds}

        names_in_three_energy_levels = names_le.intersection(names_me, names_he)
        le_compounds_in_three_energy = []
        me_compounds_in_three_energy = []
        he_compounds_in_three_energy = []

        last_num_peaks_le = 0
        last_num_peaks_me = 0
        last_num_peaks_he = 0

        for name in names_in_three_energy_levels:
            for d in le_compounds:
                if d["NAME"] == name and float(d["Num Peaks"])> last_num_peaks_le:
                    last_le_compound = d
                    last_num_peaks_le = float(d["Num Peaks"])
            if last_num_peaks_le > 0:
                le_compounds_in_three_energy.append(last_le_compound)
                last_num_peaks_le = 0
            for d in me_compounds:
                if d["NAME"] == name and float(d["Num Peaks"])> last_num_peaks_me:
                    last_me_compound = d
                    last_num_peaks_me = float(d["Num Peaks"])
            if last_num_peaks_me > 0:
                me_compounds_in_three_energy.append(last_me_compound)
                last_num_peaks_me = 0
            for d in he_compounds:
                if d["NAME"] == name and float(d["Num Peaks"])> last_num_peaks_he:
                    last_he_compounds = d
                    last_num_peaks_he = float(d["Num Peaks"])
            if last_num_peaks_he > 0:
                he_compounds_in_three_energy.append(last_he_compounds)
                last_num_peaks_he = 0
        if save_info:
            DB_INFO = {"NumberOfCompoundsIn3EnergyLevels": len(names_in_three_energy_levels), 
                    "NumberOfCompoundsInAllDB": total_compounds_count,
                    "NumberOfCompoundsHaveEnergyInfo": total_compounds_have_energy_info}

            np.save(os.path.join(filepath, "info_databases"), DB_INFO)

        return le_compounds_in_three_energy, me_compounds_in_three_energy, he_compounds_in_three_energy

    def load_data_in_ionmodes_from_databases(self, filepath="/home/julian/Documents/Librerias publicas/", save_info=False):
        """
        This method loads all the compounds available in the files allocated under `filepath`, and only returns the compounds for which there is a mass spectra in each of the two ion modes. 

        These files correspond to database `.msp` files containing information of the compound such as its name, precursormz, mass spectra, smiles, inkikey, among others. 

        There are repeated compounds. The criterion to filter repeated compounds (choose one of them) is to pick the one having the more data in the mass spectra.

        Args:
            filepath (str, optional): path to the directory under which are allocated all the files containing compounds information. Defaults to "/home/julian/Documents/Librerias publicas/".
            save_info (bool, optional): if true saves some useful information about all the scraped database files. Defaults to False.

        Returns:
            ionmode_positive_compounds_in_both_iomodes (list of dicts): each entry of the list is a dictionary having information of the compound such as its name, smiles, inkikey and mass spectra for a positive ion mode.
            ionmode_negative_compounds_in_both_iomodes (list of dicts): each entry of the list is a dictionary having information of the compound such as its name, smiles, inkikey and mass spectra for a negative ion mode.
        """

        ionmode_positive_compounds = []
        ionmode_negative_compounds = []

        total_compounds_count = 0
        total_compounds_have_ionmode_info = 0

        for file_i in os.listdir(filepath):
            aux = file_i.split(".")
            if aux[1] == 'npy':
                this_database = np.load(os.path.join(filepath, file_i), allow_pickle=True)
                for entry in this_database:
                    for key, val in entry.items():
                        if key == "IONMODE" and val != '' and val != "--":
                            if val == "Positive":
                                ionmode_positive_compounds.append(entry)
                            if val == "Negative":
                                ionmode_negative_compounds.append(entry)
                            else:
                                continue

                            total_compounds_have_ionmode_info = total_compounds_have_ionmode_info + 1

                    total_compounds_count = total_compounds_count + 1

        names_ionmode_positive = {d["NAME"] for d in ionmode_positive_compounds}
        names_ionmode_negative = {d["NAME"] for d in ionmode_negative_compounds}

        names_in_both_ionmode = names_ionmode_positive.intersection(names_ionmode_negative)

        ionmode_positive_compounds_in_both_iomodes = []
        ionmode_negative_compounds_in_both_iomodes = []

        last_num_peaks_ionmode_positive = 0
        last_num_peaks_ionmode_negative = 0

        for name in names_in_both_ionmode:
            for d in ionmode_positive_compounds:
                if d["NAME"] == name and float(d["Num Peaks"])> last_num_peaks_ionmode_positive:
                    last_ionmode_positive_compound = d
                    last_num_peaks_ionmode_positive = float(d["Num Peaks"])
            if last_num_peaks_ionmode_positive > 0:
                ionmode_positive_compounds_in_both_iomodes.append(last_ionmode_positive_compound)
                last_num_peaks_ionmode_positive = 0
            for d in ionmode_negative_compounds:
                if d["NAME"] == name and float(d["Num Peaks"])> last_num_peaks_ionmode_negative:
                    last_ionmode_negative_compound = d
                    last_num_peaks_ionmode_negative = float(d["Num Peaks"])
            if last_num_peaks_ionmode_negative > 0:
                ionmode_negative_compounds_in_both_iomodes.append(last_ionmode_negative_compound)
                last_num_peaks_ionmode_negative = 0
        if save_info:
            DB_INFO = {"NumberOfCompoundsInBothIonModes": len(names_in_both_ionmode), 
                    "NumberOfCompoundsInAllDB": total_compounds_count,
                    "NumberOfCompoundsHaveIonModeInfo": total_compounds_have_ionmode_info}

            np.save(os.path.join(filepath, "info_databases_ionmode"), DB_INFO)

        return ionmode_positive_compounds_in_both_iomodes, ionmode_negative_compounds_in_both_iomodes    

    def load_all_compounds_database(self, filepath="/home/julian/Documents/Librerias publicas/", min_num_peaks=3):
        """
        Concatenates in a list the compounds of (possibly) multiple pickle files, each of them having molecular information for multiple compounds.

        The pickle files are all under the directory `filepath`.

        Args:
            filepath (str, optional): path to directory under which are located all the pickle files with compound info. Defaults to "/home/julian/Documents/Librerias publicas/".
            min_num_peaks (int, optional): only save compounds having mass spectra with a minimum number of peaks given by this parameter. Defaults to 3.

        Returns:
            _type_: _description_
        """
        compounds = []

        for file_i in os.listdir(filepath):
            aux = file_i.split(".")
            if aux[1] == 'npy':
                this_database = np.load(os.path.join(filepath, file_i), allow_pickle=True)
                for entry in this_database:
                    for key, val in entry.items():
                        if key == "Num Peaks" and float(val) >= min_num_peaks:
                            compounds.append(entry)

        return compounds

    def load_compounds_from_file(self, filepath, idx=None):
        """
        Returns the m/z, intensities and smiles contained in the pickle file `filepath`.

        The file might contain m/z, intensities, smiles for more than 1 compound.

        Args:
            filepath (str): path to the pickle file
            idx (int, optional): the index of a particular molecule from the pickle file. Defaults to None.

        Returns:
            INT (list of lists): each sublist has the intensities (relative abundance) for each compound in the file.
            MZ (list of lists): each sublist has the m/z for each compound in the file.
            SMILES (list): each entry has the SMILES for each compound in the file.
        """
        molecules = np.load(filepath, allow_pickle=True)

        if idx is not None:
            mol_of_interest = []

            for i in idx:
                mol_of_interest.append(molecules[i])
        else:
            mol_of_interest = molecules

        INT = []
        MZ = []
        SMILES = []

        for mol in mol_of_interest:
            intensities = []
            masscharges = []
            for i in mol['peaks']:
                intensities.append(i[1])
                masscharges.append(i[0])

            SMILES.append(mol['SMILES'])
            INT.append(intensities)
            MZ.append(masscharges)

        return INT, MZ, SMILES

    def get_mz_and_int_from_list(self, list_compounds):
        """
        Gets the mass-charge and intensities values from a list of compounds of the type produced by `load_data_in_three_energy_levels_from_databases` or `load_data_in_ionmodes_from_databases`.

        Args:
            list_compounds (list of dicts): each entry of the list is a dictionary having information of the compound such as its name, smiles, inkikey and mass spectra 

        Returns:
            MZ (list of lists): each sub-list is for a compound, and has the mass-charge values of the compound
            INT (list of lists): each sub-list is for a compound, and has the intensity values of the compound
        """
        MZ = []
        INT = []
        for compound_dict in list_compounds:
            peaks = compound_dict["peaks"]
            this_mz = [x[0] for x in peaks]
            this_int = [x[1] for x in peaks]
            MZ.append(this_mz)
            INT.append(this_int)

        return MZ, INT

    def get_mz_and_smiles_from_list(self, list_compounds):
        """
        Gets the m/z and SMILES from a list of compounds of the type produced by `load_data_in_three_energy_levels_from_databases` or `load_data_in_ionmodes_from_databases`.

        Args:
            list_compounds (list of dicts): each entry of the list is a dictionary having information of the compound such as its name, smiles, inkikey and mass spectra

        Returns:
            dataset (list of dicts): each dictionary of the list represents a compound, and has two keys: "smiles" and "mz", for the SMILES of the compound and its mass-charge values, respectively
            precursormz (list of dicts): each dictionary of the list represents a compound and has two keys: "smiles" and "precursormz", for the SMILES of the compound and its precursor mass
        """

        dataset = []
        precursormz = []
        for compound_dict in list_compounds:
            peaks = compound_dict["peaks"]
            this_mz = [x[0] for x in peaks]
            dataset.append({"smiles": compound_dict["SMILES"], "mz": this_mz})
            precursormz.append({"smiles": compound_dict["SMILES"], "precursormz": compound_dict["PRECURSORMZ"]})
        return dataset, precursormz

    """
    Returns the m/z, intensities and smiles contained in the pickle file `filepath`.

    The file might contain m/z, intensities, smiles for more than 1 compound.

    MZ = []

            masscharges.append(i[0])

    class MoleculesAndSpectraDataset(Dataset):
    def __init__(self, sentences, 
                 tokenizer, 
                 max_mz=500, 
                 target_length=100, 
                 pad_data_aug=True, 
                 max_outputs=20,
                 additional_features=True, 
                 precursor_mz=False):
        super(MoleculesAndSpectraDataset, self).__init__()
        self.additional_features = additional_features
        self.precursor_mz = precursor_mz
        self.sentences = sentences
        self.tokenizer = tokenizer
        self.max_mz = max_mz
        self.target_length = target_length
        self.max_outputs = max_outputs
        self.pad_data_aug = pad_data_aug

    def create_random_padded_versions(self, original_list):
        # This is guaranteed by the caller
        #if len(original_list) > self.target_length:
        #    return []

        padded_versions = []
        original_length = len(original_list)
        max_indices = self.target_length - original_length + 1
        possible_start_indices = random.sample(range(max_indices), 
                                               min(self.max_outputs, max_indices))

        for start_idx in possible_start_indices:
            padded_list = [self.tokenizer.vocab_smi['<PAD>']] * self.target_length
            padded_list[start_idx:start_idx + original_length] = original_list
            padded_versions.append(padded_list)

        return padded_versions

    def add_features(self, mz_input):
        distances = pdist(np.array(mz_input).reshape(-1, 1), metric='euclidean')

        # Crop the distances
        distances = distances[:100]
        return distances

    def process_unknown_tokens(self, token):
        """
        Process tokens that where not found in the tokenizer vocabulary. Those tokens are split into 'character-level' tokens.

        THE BIGGEST SIZE OF A 'CHARACTER-LEVEL' TOKEN IS 2 CHARACTERS.

        Args:
            token (_type_): _description_

        Returns:
            _type_: _description_
        """
        alive = True
        token_enc = []
        while(alive):
            if token[:2] not in self.tokenizer.vocab_smi:
                indexes_to_remove = [0]
                token_enc.append(self.tokenizer.vocab_smi[token[0]])
            else:
                indexes_to_remove = [0,1]
                token_enc.append(self.tokenizer.vocab_smi[token[:2]])
            token = ''.join(char for i, char in enumerate(token) if i not in indexes_to_remove)
            if len(token) == 0:
                alive = False

        return token_enc

    def process_sentence(self, sentence):
        mz_input = sentence['mz']
        smiles = sentence['smiles'].strip()

        mz_input = [int(100 * float(x)) for x in mz_input]

        # Tokenize SMILES
        SS = self.tokenizer.tokenize(smiles)

        smi_input = []
        for token in SS:
            if token in self.tokenizer.vocab_smi:

    def process_unknown_tokens(self, token):
    """
    Process tokens that where not found in the tokenizer vocabulary. Those tokens are split into 'character-level' tokens.

    THE BIGGEST SIZE OF A 'CHARACTER-LEVEL' TOKEN IS 2 CHARACTERS.

    Args:
        token (_type_): _description_

    Returns:
        _type_: _description_
    """
    alive = True
    token_enc = []
    while(alive):
        if token[:2] not in self.tokenizer.vocab_smi:
            indexes_to_remove = [0]
            token_enc.append(self.tokenizer.vocab_smi[token[0]])
        else:
            indexes_to_remove = [0,1]
            token_enc.append(self.tokenizer.vocab_smi[token[:2]])
        token = ''.join(char for i, char in enumerate(token) if i not in indexes_to_remove)
        if len(token) == 0:
            alive = False

    return token_enc
