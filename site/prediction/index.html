    class PredictWorkflowModule(object):
    """
    Prediction workflow for molecular structure generation using different transformer architectures.
    Handles model loading, input preparation, decoding, and result ranking for various transformer variants.
    """
    def __init__(self, architecture='arch_1'):
        """
        Initialize PredictWorkflow with specified architecture.

        Args:
            architecture (str): Architecture to use. Options: 'arch_1', 'arch_2', 'arch_3'
                - arch_1: Basic transformer (MolecularStructureTeFT)
                - arch_2: Transformer with bias (MolecularStructureTeFTWithBias)
                - arch_3: Transformer with intensity (MolecularStructureTeFTWithIntensity)
        """
        self.architecture = architecture

        # Map architecture names to transformer classes
        self.transformer_classes = {
            'arch_1': TransformerArch1,
            'arch_2': TransformerArch2,
            'arch_3': TransformerArch3
        }

        if architecture not in self.transformer_classes:
            raise ValueError(f"Unknown architecture: {architecture}. Available options: {list(self.transformer_classes.keys())}")

    def get_transformer_class(self):
        """
        Get the appropriate transformer class for the selected architecture.

        Returns:
            class: Transformer class corresponding to the selected architecture.
        """
        return self.transformer_classes[self.architecture]

    def get_encoder_outputs(self, model, enc_input, intensities=None):
        """
        Get encoder outputs handling different architectures.

        Args:
            model: The transformer model
            enc_input: The encoded input tensor
            intensities: Optional intensities for arch_3

        Returns:
            tuple: (enc_outputs, enc_self_attns) for all architectures
        """
        if self.architecture == 'arch_1':
            enc_outputs, enc_self_attns = model.encoder(enc_input)
        elif self.architecture == 'arch_2':
            enc_outputs, enc_self_attns, raw_enc_inputs = model.encoder(enc_input)
        elif self.architecture == 'arch_3':
            enc_outputs, enc_self_attns, raw_enc_inputs, intensities = model.encoder(enc_input, intensities)
        else:
            raise ValueError(f"Unknown architecture: {self.architecture}")

        return enc_outputs, enc_self_attns

    def make_encoder_inputs(self, tokenizer, mz, prid_num=100, intensities=None):
        """
        Create encoder input tensor from m/z values.

        Args:
            tokenizer: Tokenizer object with vocab_mz attribute.
            mz (list or numpy.ndarray): List or array of m/z values.
            prid_num (int, optional): Number of positions to pad/truncate to. Defaults to 100.
            intensities (list or numpy.ndarray, optional): List or array of intensity values. Defaults to None.

        Returns:
            list: List containing the encoder input sequence as indices.
        """
        mz = [float(x) for x in mz]
        mz = [int(100 * x) for x in mz]

        ity = [float(x) for x in intensities]
        ity = [x/max(ity) for x in ity]

        print(f"mz shape: {len(mz)}, ity shape: {len(ity)}")

        #if max(mz) > 50000:
        #    return [] 

        enc_input2 = []

        for j in range(0, prid_num):
            mz.append(0)
            ity.append(0)

        print(f"mz shape after padding: {len(mz)}, ity shape after padding: {len(ity)}")

        enc_input = [tokenizer.vocab_mz[n] for n in mz]
        enc_input2.append(enc_input[0:prid_num])

        ity = [ity[0:prid_num]]

        if intensities is not None:
            return enc_input2, ity
        else:
            return enc_input2   

    def make_decoder_inputs(self, tokenizer, mz, smiles, prid_num=100):
        """
        Create decoder input tensor from SMILES string and m/z values.

        Args:
            tokenizer: Tokenizer object with vocab_smi attribute.
            mz (list): List of m/z values.
            smiles (str): SMILES string.
            prid_num (int, optional): Number of positions to pad/truncate to. Defaults to 100.

        Returns:
            tuple: (decoder input list, real_smiles list)
        """

        if max(mz) > 50000:
            return []

        judge = 0
        SS = []
        dec_input2 = []
        real_smiles = []

        for j in range(len(smiles)):
            if judge == 1:
                judge = 0
            else:
                if smiles[j] == 'C' and j < len(smiles) - 1:
                    if smiles[j + 1] == 'l':
                        SS.append('Cl')
                        judge = 1
                        continue
                if smiles[j] == 'B' and j < len(smiles) - 1:
                    if smiles[j + 1] == 'r':
                        SS.append('Br')
                        judge = 1
                        continue
                if smiles[j] == '\xa0': # detects a space at the end of the line in the string
                    continue
                SS.append(smiles[j])
        SS = ['<SOS>'] + SS + ['<EOS>']

        for j in range(0, prid_num):
            SS.append('<PAD>')

        dec_input = [tokenizer.vocab_smi[n] for n in SS]
        dec_input2.append(dec_input[0:prid_num+1])
        real_smiles.append(smiles)

        return dec_input2, real_smiles

    def greedy_decoder(self, model, enc_input, molmol, start_symbol, device, tokenizer, intensities=None):
        """
        Greedy decoding for sequence generation.

        Args:
            model: Transformer model with encoder and decoder.
            enc_input (torch.Tensor): Encoder input tensor.
            molmol (str): Target sequence for forced decoding.
            start_symbol (int): Start token index.
            device: Torch device.
            tokenizer: Tokenizer object with vocab_smi attribute.
            intensities (torch.Tensor, optional): Optional intensities for arch_3. Defaults to None.

        Returns:
            torch.Tensor: Decoded sequence tensor (without initial start symbol).
        """
        enc_outputs, enc_self_attns = self.get_encoder_outputs(model, enc_input, intensities)
        dec_input = torch.zeros(1, 0).type_as(enc_input.data)
        terminal = False
        next_symbol = start_symbol
        while not terminal:
            dec_input = torch.cat([dec_input.to(device), torch.tensor([[next_symbol]], dtype=enc_input.dtype).to(device)],
                                -1)
            dec_outputs, _, _ = model.decoder(dec_input, enc_input, enc_outputs)
            projected = model.projection(dec_outputs)
            prob = projected.squeeze(0).max(dim=-1, keepdim=False)[1]
            next_word = prob.data[-1]
            next_symbol = next_word
            # print(next_word)
            num = dec_input.shape
            if num[1] < len(molmol) + 1 and num[1] != 0:
                ana = molmol[num[1] - 1]
                next_symbol = torch.tensor(tokenizer.vocab_smi[ana]).to(device)
            if num[1] == 101:
                terminal = True
            # print(next_word)

        # greedy_dec_predict = torch.cat(
        #     [dec_input.to(device), torch.tensor([[next_symbol]], dtype=enc_input.dtype).to(device)],
        #     -1)
        greedy_dec_predict = dec_input[:, 1:]
        return greedy_dec_predict

    def load_model_and_tokenizer(self, full_path_to_mdl, device):
        """
        Loads the model and tokenizer from the checkpoint.
        Args:
            full_path_to_mdl (str): Path to model checkpoint.
            device: Torch device.
        Returns:
            model: Loaded transformer model.
            tokenizer: Loaded or default tokenizer.
        """
        checkpoint = load_model_with_compatibility(full_path_to_mdl)
        chekpoint_modified = checkpoint.copy()
        d_k = checkpoint.get('d_k', 64)
        d_v = checkpoint.get('d_v', 64)
        d_ff = checkpoint.get('d_ff', 2048)
        n_heads = checkpoint.get('n_heads', 8)
        n_layers = checkpoint.get('n_layers', 6)
        d_model = checkpoint.get('d_model', 512)
        if 'tokenizer' in checkpoint:
            tokenizer = checkpoint['tokenizer']
            chekpoint_modified.pop('tokenizer')
        else:
            tokenizer = DataDrivenTokenizer()
        # Remove used keys
        for k in ['d_k', 'd_v', 'd_ff', 'n_heads', 'n_layers', 'd_model']:
            if k in chekpoint_modified:
                chekpoint_modified.pop(k)
        TransformerClass = self.get_transformer_class()
        model = TransformerClass(d_k=d_k, 
                                d_ff=d_ff, 
                                d_model=d_model, 
                                d_v=d_v, 
                                n_heads=n_heads, 
                                n_layers=n_layers, 
                                device=device, 
                                vocab_mz_size=tokenizer.vocab_mz_size,
                                vocab_smi_size=tokenizer.vocab_smi_size).to(device)
        if 'model_state_dict' in chekpoint_modified:
            state_dict = chekpoint_modified.get('model_state_dict', chekpoint_modified)
        elif 'state_dict' in chekpoint_modified:
            state_dict = chekpoint_modified.get('state_dict', chekpoint_modified)
            state_dict = {k.replace("model.", ""): v for k, v in state_dict.items()}
        else:
            state_dict = chekpoint_modified
        model.load_state_dict(state_dict)
        return model, tokenizer

    def prepare_inputs(self, tokenizer, Spec_list_Mz, smiles=None, sample_i=None, intensities=None):
        """
        Prepares encoder and decoder inputs for a sample.
        Args:
            tokenizer: Tokenizer object.
            Spec_list_Mz: List of m/z values for the sample.
            smiles: List of ground truth SMILES strings (optional).
            sample_i: Index of the sample (optional, required if smiles is provided).
            intensities: List of intensity values for the sample (optional).
        Returns:
            enc_input2: Encoder input.
            dec_input2: Decoder input (if smiles is provided, else None).
            real_smiles: Real SMILES (if smiles is provided, else None).
        """
        if smiles is None:
            if intensities is not None: # for arch_3
                enc_input2, ity = self.make_encoder_inputs(tokenizer, Spec_list_Mz, intensities=intensities)
                if len(enc_input2) == 0:
                    return [], None, None, None
                else:
                    return enc_input2, ity, None, None
            else:
                enc_input2 = self.make_encoder_inputs(tokenizer, Spec_list_Mz)
                if len(enc_input2) == 0:
                    return [], None, None
                else:
                    return enc_input2, None, None
        else:
            dec_input2, real_smiles = self.make_decoder_inputs(tokenizer, Spec_list_Mz, smiles[sample_i])
            if intensities is not None:
                enc_input2, ity = self.make_encoder_inputs(tokenizer, Spec_list_Mz, intensities=intensities)
                if len(enc_input2) == 0:
                    return [], None, None, None
                else:
                    return enc_input2, ity, dec_input2, real_smiles
            else:
                enc_input2 = self.make_encoder_inputs(tokenizer, Spec_list_Mz)
                if len(enc_input2) == 0:
                    return [], None, None
                else:
                    return enc_input2, dec_input2, real_smiles

    def predict_single_sample(self, model, tokenizer, enc_inputs, intensities, num_pred, predict_approach, predict_approach_args, real_smiles=None):
        """
        Runs prediction/decoding for a single sample.
        Args:
            model: Transformer model.
            tokenizer: Tokenizer object.
            enc_inputs: Encoder input tensor(s).
            intensities: Intensities for the sample.
            num_pred: Number of predictions per input.
            predict_approach: Decoding approach ("greedy" or "beam_search").
            predict_approach_args: Arguments for decoding approach.
            real_smiles: Real SMILES string(s) for the sample (optional).
        Returns:
            ans: List of prediction results for this sample.
            SIM: List of similarity scores (if real_smiles is provided).
            cnt_correct_smiles: Count of correct predictions (if real_smiles is provided).
        """
        jihe = []
        SIM = []
        ans = []
        sm = ''
        cnt_correct_smiles = 0
        for i in range(len(enc_inputs)):
            for j in range(num_pred):
                if predict_approach == "greedy":
                    enc_smiles_predicted = self.greedy_decoder(model, enc_inputs[i].view(1, -1).to(enc_inputs.device), sm, start_symbol=tokenizer.vocab_smi['<SOS>'], device=enc_inputs.device, tokenizer=tokenizer, intensities=intensities)
                elif predict_approach == "beam_search":
                    enc_smiles_predicted = self.beam_search_decoder(model, enc_inputs[i].view(1, -1), tokenizer.vocab_smi['<SOS>'], beam_size=predict_approach_args['beam_size'], intensities=intensities)
                predictsmiles = [tokenizer.vocab_smi_rev[n.item()] for n in enc_smiles_predicted.squeeze()]
                psm = ''.join(predictsmiles)
                psm = psm.replace('<PAD>', '')
                c = psm.find('<EOS>')
                if real_smiles is not None:
                    realsmiles = real_smiles[i]
                    try:
                        max_sim = similarity_score(psm[0:c], str(realsmiles), which_similarity="all")
                        tomitosim = similarity_score(psm[0:c], str(realsmiles), which_similarity="tomitosim")
                    except Exception as e:
                        max_sim = 0.0
                        tomitosim = 0.0
                    SIM.append(max_sim)
                    DITT = dict(num=i, predict=psm[0:c], real=str(realsmiles), similar=max_sim, tomitosim=tomitosim)
                    print('predictsmiles:', psm[0:c])
                    print('realsmiles:', realsmiles)
                    if psm[0:c] == realsmiles:
                        cnt_correct_smiles += 1
                else:
                    DITT = dict(num=i, predict=psm[0:c])
                    print('predictsmiles:', psm[0:c])
                jihe.append(DITT)
            if real_smiles is not None:
                if SIM:
                    max_sim = max(SIM)
                    pos = SIM.index(max_sim)
                    print("--------------------------------------")
                    print('From all predicted smiles, max sim:', max_sim)
                    print('Smiles for max sim:', jihe[pos]['predict'])
            ans.append(jihe)
            SIM = []
            jihe = []
        return ans, cnt_correct_smiles

    def write_predictions_to_file(self, ans, smiles, filenames, sample_i):
        """
        Writes the predictions for a sample to a file, sorting as needed.
        Args:
            ans: List of prediction results for this sample.
            smiles: List of ground truth SMILES strings (optional).
            filenames: List of output filenames.
            sample_i: Index of the sample.
        """
        if smiles is not None:
            sorted_data = sorted(ans[0], key=lambda x: x['similar'], reverse=True)
        else:
            sorted_data = self.ranking_smiles(ans[0])
        final = json.dumps(sorted_data)
        with open(filenames[sample_i], 'w') as fileOb:
            fileOb.write(final)
        print('predict!')

    def predict(self, INT, 
                MZ, 
                smiles=None, 
                full_path_to_mdl="/home", 
                device=None, 
                output_dir="", 
                ids_ms2=None, 
                num_pred=10, 
                predict_approach="greedy",
                predict_approach_args=None):
        """
        Run prediction for a batch of spectra and optionally compare to ground truth SMILES.
        Args:
            INT (list of lists): Intensities for one or more MS2 spectra.
            MZ (list of lists): m/z values for one or more MS2 spectra.
            smiles (list, optional): List of ground truth SMILES strings. Defaults to None.
            full_path_to_mdl (str, optional): Path to model checkpoint. Defaults to "/home".
            device: Torch device. Defaults to None.
            output_dir (str, optional): Directory to save results. Defaults to "".
            ids_ms2 (list, optional): List of MS2 identifiers. Defaults to None.
            num_pred (int, optional): Number of predictions per input. Defaults to 10.
            predict_approach (str, optional): Decoding approach ("greedy" or "beam_search"). Defaults to "greedy".
            predict_approach_args (dict, optional): Arguments for decoding approach. Defaults to None.
        Returns:
            None
        """
        my_signal_math = SignalMath()
        if device is None:
            device = self.device
        if not os.path.exists(output_dir):
            os.makedirs(output_dir)
        if ids_ms2 is not None:
            filenames = [f'{output_dir}/pred_{ids_ms2[i]}.json' for i in range(len(MZ))]
        else:
            filenames = [f'{output_dir}/pred_{i}.json' for i in range(len(MZ))]
        model, tokenizer = self.load_model_and_tokenizer(full_path_to_mdl, device)
        cnt_correct_smiles = 0
        for sample_i in range(len(MZ)):
            intensities, Spec_list_Mz = my_signal_math.find_peak(np.array(INT[sample_i]), np.array(MZ[sample_i]))

            if intensities is not None:
                enc_input2, intensities, dec_input2, real_smiles = self.prepare_inputs(tokenizer, Spec_list_Mz, smiles, sample_i, intensities=intensities)
            else:
                enc_input2, dec_input2, real_smiles = self.prepare_inputs(tokenizer, Spec_list_Mz, smiles, sample_i)

            if len(enc_input2) == 0:
                continue

            enc_inputs = torch.LongTensor(enc_input2).to(device)

            if intensities is not None:
                intensities = torch.LongTensor(intensities).to(torch.float32).to(device)

            ans, correct_count = self.predict_single_sample(
                model, tokenizer, enc_inputs, intensities, num_pred, predict_approach, predict_approach_args, real_smiles=real_smiles if smiles is not None else None
            )
            cnt_correct_smiles += correct_count
            self.write_predictions_to_file(ans, smiles, filenames, sample_i)
        print(f"Accuracy: {cnt_correct_smiles/len(MZ)}")

        # Simple ranking - could be enhanced with other metrics

    def beam_search_decoder(self, model, enc_input, start_symbol, beam_size=5, max_len=101, intensities=None):
        """
        Beam search decoder for sequence generation.

        Args:
            model: The transformer model.
            enc_input (torch.Tensor): The encoded input tensor.
            start_symbol (int): The starting token for decoding.
            beam_size (int, optional): Number of sequences to keep in the beam. Defaults to 5.
            max_len (int, optional): Maximum sequence length. Defaults to 101.
            intensities (torch.Tensor, optional): Optional intensities for arch_3. Defaults to None.

        Returns:
            torch.Tensor: The best decoded sequence (without initial start symbol).
        """
        enc_outputs, _ = self.get_encoder_outputs(model, enc_input, intensities)
        device = enc_input.device

        # Initialize beam with start_symbol
        beams = [(torch.tensor([[start_symbol]], dtype=enc_input.dtype, device=device), 0.0)]  # (sequence, score)

        for _ in range(max_len):
            candidates = []
            for seq, score in beams:
                dec_input = seq.to(device)
                dec_outputs, _, _ = model.decoder(dec_input, enc_input, enc_outputs)
                projected = model.projection(dec_outputs)  # Get logits

                # Get the last token probabilities
                log_probs = F.log_softmax(projected[:, -1, :], dim=-1)  # (1, vocab_size)

                # Get top beam_size candidates
                top_k_probs, top_k_words = torch.topk(log_probs, beam_size, dim=-1)

                for i in range(beam_size):
                    next_token = top_k_words[0, i].item()
                    next_score = score + top_k_probs[0, i].item()

                    # Extend the sequence

    def beam_search_decoder(self, model, enc_input, start_symbol, beam_size=5, max_len=101, intensities=None):
    """
    Beam search decoder for sequence generation.

    Args:
        model: The transformer model.
        enc_input (torch.Tensor): The encoded input tensor.
        start_symbol (int): The starting token for decoding.
        beam_size (int, optional): Number of sequences to keep in the beam. Defaults to 5.
        max_len (int, optional): Maximum sequence length. Defaults to 101.
        intensities (torch.Tensor, optional): Optional intensities for arch_3. Defaults to None.

    Returns:
        torch.Tensor: The best decoded sequence (without initial start symbol).
    """
    enc_outputs, _ = self.get_encoder_outputs(model, enc_input, intensities)
    device = enc_input.device

    # Initialize beam with start_symbol
    beams = [(torch.tensor([[start_symbol]], dtype=enc_input.dtype, device=device), 0.0)]  # (sequence, score)

    for _ in range(max_len):
        candidates = []
        for seq, score in beams:
            dec_input = seq.to(device)
            dec_outputs, _, _ = model.decoder(dec_input, enc_input, enc_outputs)
            projected = model.projection(dec_outputs)  # Get logits

            # Get the last token probabilities
            log_probs = F.log_softmax(projected[:, -1, :], dim=-1)  # (1, vocab_size)

            # Get top beam_size candidates
            top_k_probs, top_k_words = torch.topk(log_probs, beam_size, dim=-1)

            for i in range(beam_size):
                next_token = top_k_words[0, i].item()
                next_score = score + top_k_probs[0, i].item()

                # Extend the sequence
                new_seq = torch.cat([seq, torch.tensor([[next_token]], dtype=seq.dtype, device=device)], dim=-1)
                candidates.append((new_seq, next_score))

        # Sort candidates by score and keep top `beam_size`
        beams = sorted(candidates, key=lambda x: x[1], reverse=True)[:beam_size]

        # Check if all beams have reached the max length
        if all(seq.shape[1] >= max_len for seq, _ in beams):
            break

    # Return the best sequence (highest score)
    best_seq, _ = max(beams, key=lambda x: x[1])
    return best_seq[:, 1:]  # Remove initial start symbol

    def greedy_decoder(self, model, enc_input, molmol, start_symbol, device, tokenizer, intensities=None):
    """
    Greedy decoding for sequence generation.

    Args:
        model: Transformer model with encoder and decoder.
        enc_input (torch.Tensor): Encoder input tensor.
        molmol (str): Target sequence for forced decoding.
        start_symbol (int): Start token index.
        device: Torch device.
        tokenizer: Tokenizer object with vocab_smi attribute.
        intensities (torch.Tensor, optional): Optional intensities for arch_3. Defaults to None.

    Returns:
        torch.Tensor: Decoded sequence tensor (without initial start symbol).
    """
    enc_outputs, enc_self_attns = self.get_encoder_outputs(model, enc_input, intensities)
    dec_input = torch.zeros(1, 0).type_as(enc_input.data)
    terminal = False
    next_symbol = start_symbol
    while not terminal:
        dec_input = torch.cat([dec_input.to(device), torch.tensor([[next_symbol]], dtype=enc_input.dtype).to(device)],
                            -1)
        dec_outputs, _, _ = model.decoder(dec_input, enc_input, enc_outputs)
        projected = model.projection(dec_outputs)
        prob = projected.squeeze(0).max(dim=-1, keepdim=False)[1]
        next_word = prob.data[-1]
        next_symbol = next_word
        # print(next_word)
        num = dec_input.shape
        if num[1] < len(molmol) + 1 and num[1] != 0:
            ana = molmol[num[1] - 1]
            next_symbol = torch.tensor(tokenizer.vocab_smi[ana]).to(device)
        if num[1] == 101:
            terminal = True
        # print(next_word)

    # greedy_dec_predict = torch.cat(
    #     [dec_input.to(device), torch.tensor([[next_symbol]], dtype=enc_input.dtype).to(device)],
    #     -1)
    greedy_dec_predict = dec_input[:, 1:]
    return greedy_dec_predict

    for k in ['d_k', 'd_v', 'd_ff', 'n_heads', 'n_layers', 'd_model']:
        if k in chekpoint_modified:
            chekpoint_modified.pop(k)
    TransformerClass = self.get_transformer_class()
    model = TransformerClass(d_k=d_k, 
                            d_ff=d_ff, 
                            d_model=d_model, 
                            d_v=d_v,

                            n_layers=n_layers,

    return model, tokenizer

    def make_decoder_inputs(self, tokenizer, mz, smiles, prid_num=100):
    """
    Create decoder input tensor from SMILES string and m/z values.

    Args:
        tokenizer: Tokenizer object with vocab_smi attribute.
        mz (list): List of m/z values.
        smiles (str): SMILES string.
        prid_num (int, optional): Number of positions to pad/truncate to. Defaults to 100.

    Returns:
        tuple: (decoder input list, real_smiles list)
    """

    if max(mz) > 50000:
        return []

    judge = 0
    SS = []
    dec_input2 = []
    real_smiles = []

    for j in range(len(smiles)):
        if judge == 1:
            judge = 0
        else:
            if smiles[j] == 'C' and j < len(smiles) - 1:
                if smiles[j + 1] == 'l':
                    SS.append('Cl')
                    judge = 1
                    continue
            if smiles[j] == 'B' and j < len(smiles) - 1:
                if smiles[j + 1] == 'r':
                    SS.append('Br')
                    judge = 1
                    continue
            if smiles[j] == '\xa0': # detects a space at the end of the line in the string
                continue
            SS.append(smiles[j])
    SS = ['<SOS>'] + SS + ['<EOS>']

    for j in range(0, prid_num):
        SS.append('<PAD>')

    dec_input = [tokenizer.vocab_smi[n] for n in SS]
    dec_input2.append(dec_input[0:prid_num+1])
    real_smiles.append(smiles)

    return dec_input2, real_smiles

                enc_smiles_predicted = self.greedy_decoder(model, enc_inputs[i].view(1, -1).to(enc_inputs.device), sm, start_symbol=tokenizer.vocab_smi['<SOS>'], device=enc_inputs.device, tokenizer=tokenizer, intensities=intensities)
            elif predict_approach == "beam_search":
                enc_smiles_predicted = self.beam_search_decoder(model, enc_inputs[i].view(1, -1), tokenizer.vocab_smi['<SOS>'], beam_size=predict_approach_args['beam_size'], intensities=intensities)
            predictsmiles = [tokenizer.vocab_smi_rev[n.item()] for n in enc_smiles_predicted.squeeze()]
            psm = ''.join(predictsmiles)
            psm = psm.replace('<PAD>', '')
            c = psm.find('<EOS>')
            if real_smiles is not None:
                realsmiles = real_smiles[i]
                try:
                    max_sim = similarity_score(psm[0:c], str(realsmiles), which_similarity="all")
                    tomitosim = similarity_score(psm[0:c], str(realsmiles), which_similarity="tomitosim")
                except Exception as e:
                    max_sim = 0.0
                    tomitosim = 0.0
                SIM.append(max_sim)
                DITT = dict(num=i, predict=psm[0:c], real=str(realsmiles), similar=max_sim, tomitosim=tomitosim)
                print('predictsmiles:', psm[0:c])
                print('realsmiles:', realsmiles)
                if psm[0:c] == realsmiles:
                    cnt_correct_smiles += 1
            else:
                DITT = dict(num=i, predict=psm[0:c])
                print('predictsmiles:', psm[0:c])
            jihe.append(DITT)
        if real_smiles is not None:
            if SIM:
                max_sim = max(SIM)
                pos = SIM.index(max_sim)
                print("--------------------------------------")
                print('From all predicted smiles, max sim:', max_sim)
                print('Smiles for max sim:', jihe[pos]['predict'])
        ans.append(jihe)
        SIM = []
        jihe = []
    return ans, cnt_correct_smiles

    def prepare_inputs(self, tokenizer, Spec_list_Mz, smiles=None, sample_i=None, intensities=None):
    """
    Prepares encoder and decoder inputs for a sample.
    Args:
        tokenizer: Tokenizer object.
        Spec_list_Mz: List of m/z values for the sample.
        smiles: List of ground truth SMILES strings (optional).
        sample_i: Index of the sample (optional, required if smiles is provided).
        intensities: List of intensity values for the sample (optional).
    Returns:
        enc_input2: Encoder input.
        dec_input2: Decoder input (if smiles is provided, else None).
        real_smiles: Real SMILES (if smiles is provided, else None).
    """
    if smiles is None:
        if intensities is not None: # for arch_3
            enc_input2, ity = self.make_encoder_inputs(tokenizer, Spec_list_Mz, intensities=intensities)
            if len(enc_input2) == 0:
                return [], None, None, None
            else:
                return enc_input2, ity, None, None
        else:
            enc_input2 = self.make_encoder_inputs(tokenizer, Spec_list_Mz)
            if len(enc_input2) == 0:
                return [], None, None
            else:
                return enc_input2, None, None
    else:
        dec_input2, real_smiles = self.make_decoder_inputs(tokenizer, Spec_list_Mz, smiles[sample_i])
        if intensities is not None:
            enc_input2, ity = self.make_encoder_inputs(tokenizer, Spec_list_Mz, intensities=intensities)
            if len(enc_input2) == 0:
                return [], None, None, None
            else:
                return enc_input2, ity, dec_input2, real_smiles
        else:
            enc_input2 = self.make_encoder_inputs(tokenizer, Spec_list_Mz)
            if len(enc_input2) == 0:
                return [], None, None
            else:
                return enc_input2, dec_input2, real_smiles
